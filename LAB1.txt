#!/bin/bash

echo "================================================"
echo "This is a General System Report"
echo "================================================ "

echo "Below Represents the hostname"
echo "================================================ "
echo $HOSTNAME

echo "This is the IPaddress"
echo "====================================================== "

echo "This shows the free memory usage in the system"
echo "==============================================================================================================="

echo "================================================ "
echo $(grep -c ^processor /proc/cpuinfo)
